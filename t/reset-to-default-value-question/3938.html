@Override
public void applyDefaultValue() {
if (((ParameterFrac32) parameter).DefaultValue != null) {
value.setRaw(((ParameterFrac32) parameter).DefaultValue.getRaw());
} else {
value.setRaw(0);
}
updateV();
needsTransmit = true;
}
